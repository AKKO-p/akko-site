SELECT account_id_masked,
       SUM(amount) AS total_eur,
       COUNT(*) AS txns
FROM   banking.fact_txn

  AND  ts > now() - INTERVAL '7 days'
GROUP BY account_id_masked
ORDER BY total_eur DESC
